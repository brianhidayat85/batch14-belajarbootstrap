Looping / Perulangan 

1. For 
    format: 
    for (let iterasi=nilai_awal; iterasi < nilai_akhir; iterasi++) {
        // do something
    }

2. While Do 
    format: 
    let iterasi = awal;
    while (iterasi < nilai_akhir) {
        // do something
        iterasi++;
    }

3. Do While 
    format: 
    let iterasi = awal;
    do {
        // do something
        iterasi++;
    }
    while (iterasi < nilai_akhir);
